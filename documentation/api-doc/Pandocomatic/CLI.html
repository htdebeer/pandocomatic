def self.parse(args)
  args = args.split if args.is_a? String

  begin
    options = parse_options args || {:help => true, :help_given => true}
    options
  rescue Trollop::CommandlineError => e
    raise CLIError.new(:problematic_invocation, e, args)
  end
end
